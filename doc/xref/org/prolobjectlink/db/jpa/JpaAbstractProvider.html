81  	protected final DatabaseUnitInfo getPersistenceUnitInformation(PersistenceUnitInfo info) {
82  		String unitName = info.getPersistenceUnitName();
83  		URL unitRootUrl = info.getPersistenceUnitRootUrl();
84  		String versionXml = info.getPersistenceXMLSchemaVersion();
85  		PersistenceSchemaVersion xmlVersion = new PersistenceSchemaVersion(versionXml, "UTF-8");
86  		String transactionType = info.getTransactionType().toString();
87  		String xmlPersistenceVersion = "2.0";
88  		String xmlPersistenceXmlns = "http://java.sun.com/xml/ns/persistence";
89  		String xmlPersistenceXmlnsXsi = "http://www.w3.org/2001/XMLSchema-instance";
90  		String xmlPersistenceXsiSchemalocation = "http://java.sun.com/xml/ns/persistence http://java.sun.com/xml/ns/persistence/persistence_2_0.xsd";
91  		PersistenceVersion persistenceVersion = new PersistenceVersion(xmlPersistenceVersion, xmlPersistenceXmlns,
92  				xmlPersistenceXmlnsXsi, xmlPersistenceXsiSchemalocation);
93  		PersistenceUnitInformation unit = new PersistenceUnitInformation(unitRootUrl, xmlVersion, persistenceVersion,
94  				unitName, transactionType);
95  
96  		// TODO add all persistence unit information
97  		unit.setValidationMode(info.getValidationMode().toString());

102 	private DatabaseEngine createDatabaseEngine(DatabaseUnitInfo info, Map<?, ?> map) {
103 		try {
104 			DatabaseEngine database = null;
105 			String urlString = "" + info.getProperties().get(URL) + "";
106 			urlString = urlString.replace(URL_PREFIX, "");
107 			String url = new URL(urlString).getProtocol().toUpperCase();
108 			if (url.equals("" + Protocol.MEMPDB + "")) {
109 				database = MemoryHierarchical.newInstance(info, map);
110 			} else if (url.equals("" + Protocol.REMPDB + "")) {
111 				database = RemoteHierarchical.newInstance(info, map);
112 			} else if (url.equals("" + Protocol.FILE + "")) {
113 				database = EmbeddedHierarchical.newInstance(info, map);
114 			}
115 			return database;
116 		} catch (MalformedURLException e) {
117 			LoggerUtils.error(getClass(), e);
118 		}
119 		return null;
120 	}

129 	public final EntityManagerFactory createEntityManagerFactory(String emName, Map map) {
130 		EntityManagerFactory emf = entityManagerFactories.get(emName);
131 		if (emf == null) {
132 			ClassLoader loader = Thread.currentThread().getContextClassLoader();
133 			Map<String, DatabaseUnitInfo> m = p.parsePersistenceXml(loader.getResource(XML));
134 			assertPersistenceUnitExistenceInMapLoadedFromXml(emName, m);
135 			emf = createContainerEntityManagerFactory(m.get(emName), map);
136 		}
137 		return emf;
138 	}

158 	public final boolean generateSchema(String persistenceUnitName, Map map) {
159 		ClassLoader loader = Thread.currentThread().getContextClassLoader();
160 		Map<String, DatabaseUnitInfo> m = p.parsePersistenceXml(loader.getResource(XML));
161 		assertPersistenceUnitExistenceInMapLoadedFromXml(persistenceUnitName, m);
162 		DatabaseUnitInfo unit = m.get(persistenceUnitName);
163 		DatabaseEngine database = generateSchema0(unit, map);
164 		return database.exist();
