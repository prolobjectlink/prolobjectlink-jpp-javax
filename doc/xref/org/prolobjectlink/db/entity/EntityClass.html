132 	public String generate() {
133 		StringBuilder buffer = new StringBuilder();
134 		buffer.append("import javax.persistence.Entity");
135 		buffer.append('\n');
136 		buffer.append("@Entity");

143 		buffer.append(' ');
144 		if (superClass != null) {
145 			buffer.append("extends");
146 			buffer.append(' ');
147 			buffer.append(superClass);
148 		}
149 		Iterator<DatabaseClass> i = superClasses.iterator();
150 		if (i.hasNext()) {
151 			buffer.append("implements");
152 			buffer.append(' ');
153 		}
154 		while (i.hasNext()) {
155 			DatabaseClass clazz = i.next();
156 			buffer.append(clazz);
157 			if (i.hasNext()) {
158 				buffer.append(',');
159 				buffer.append(' ');
160 			}
161 		}
162 		buffer.append(' ');
163 		buffer.append('{');
164 		buffer.append('\n');
165 		buffer.append('\n');
166 
167 		// fields
168 		EntityField[] e = new EntityField[0];
169 		EntityField[] a = fields.values().toArray(e);
170 		Arrays.sort(a, new FieldComparator());
171 		for (EntityField field : a) {
172 			field.generateField(buffer);
173 		}
174 		buffer.append('\n');
175 
176 		// empty constructor
177 		buffer.append('\t');
178 		buffer.append(Modifier.PUBLIC);
179 		buffer.append(' ');
180 		buffer.append(shortName);
181 		buffer.append("()");
182 		buffer.append('{');
183 		buffer.append('\n');
184 		buffer.append('\t');
185 		buffer.append('\t');
186 		buffer.append('\n');
187 		buffer.append('\t');
188 		buffer.append('}');
189 		buffer.append('\n');
190 		buffer.append('\n');
191 
192 		// full constructor
193 		buffer.append('\t');
194 		buffer.append(Modifier.PUBLIC);
195 		buffer.append(' ');
196 		buffer.append(shortName);
197 		buffer.append('(');
198 
199 		// TODO respect declaration order resolve this doing field a persistent linked
200 		// hash map
201 
202 		Iterator<DatabaseField> j = fields.values().iterator();
203 		while (j.hasNext()) {
204 			DatabaseField field = j.next();
205 			field.generateParameter(buffer);
206 			if (j.hasNext()) {
207 				buffer.append(',');
208 				buffer.append(' ');
209 			}
210 		}
211 		buffer.append(')');
212 		buffer.append('{');
213 		buffer.append('\n');
214 		for (EntityField field : a) {
215 			field.generateAssigment(buffer);
216 		}
217 		buffer.append('\t');
218 		buffer.append('}');
219 		buffer.append('\n');
220 		buffer.append('\n');
221 
222 		// methods
223 		for (DatabaseField field : fields.values()) {
224 			field.generateGetter(buffer);
225 			field.generateSetter(buffer);
226 		}
227 		buffer.append('}');
228 		buffer.append('\n');
229 		return "" + buffer + "";
230 	}
