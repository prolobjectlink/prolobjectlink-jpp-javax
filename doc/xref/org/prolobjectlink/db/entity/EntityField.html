60  		buffer.append('\t');

62  		buffer.append(' ');

64  		if (hasLinkedTypeName()) {
65  			buffer.append('<');
66  			buffer.append(getLinkedTypeShortName());

72  		buffer.append('\n');

75  	/**
76  	 * Create a field in byte code instruction
77  	 * 
78  	 * @param cv class writer to field declaration
79  	 * @since 1.0
80  	 */
81  	@Override
82  	public void createField(ClassVisitor cv) {
83  		FieldVisitor fv = null;
84  		if (isList(getType()) || isSet(getType()) || isCollection(getType())) {
85  
86  			SignatureVisitor sv = new SignatureWriter();
87  			SignatureVisitor psv = sv.visitParameterType();
88  			psv.visitClassType(Type.getInternalName(List.class));
89  			SignatureVisitor ppsv = psv.visitTypeArgument('=');
90  			ppsv.visitClassType(Type.getInternalName(String.class));
91  			ppsv.visitEnd();
92  			// psv.visitEnd();
93  			sv.visitEnd();
94  
95  			// remove not valid parenthesis at start
96  			String sig = sv.toString().substring(1);
97  
98  			fv = cv.visitField(Opcodes.ACC_PRIVATE, getName(), Type.getDescriptor(getType()), sig, null);
99  		} else {
100 			fv = cv.visitField(Opcodes.ACC_PRIVATE, getName(), Type.getDescriptor(getType()), null, null);
101 		}
102 
103 		if (isPrimaryKey()) {
104 			fv.visitAnnotation(Type.getDescriptor(Id.class), true).visitEnd();
105 			fv.visitAnnotation(Type.getDescriptor(GeneratedValue.class), true).visitEnd();
106 		}
107 		if (isOneToOne()) {
108 			fv.visitAnnotation(Type.getDescriptor(OneToOne.class), true).visitEnd();
109 		}
110 		if (isOneToMany()) {
111 			fv.visitAnnotation(Type.getDescriptor(OneToMany.class), true).visitEnd();
112 		}
113 		if (isManyToOne()) {
114 			fv.visitAnnotation(Type.getDescriptor(ManyToOne.class), true).visitEnd();
115 		}
116 		if (isManyToMany()) {
117 			fv.visitAnnotation(Type.getDescriptor(ManyToMany.class), true).visitEnd();
118 		}
119 		fv.visitEnd();
120 	}

139 	public void generateSetter(StringBuilder buffer) {
140 		String fieldName = getName();
141 		char n = Character.toUpperCase(fieldName.charAt(0));
142 		String fname = n + fieldName.substring(1);
143 		buffer.append('\t');
144 		buffer.append(Modifier.PUBLIC);
145 		buffer.append(' ');
146 		buffer.append("void");
147 		buffer.append(' ');
148 		buffer.append("set");
149 		buffer.append(fname);
150 		buffer.append('(');
151 		buffer.append(getType().getSimpleName());
152 		if (hasLinkedTypeName()) {
153 			buffer.append('<');
154 			buffer.append(getLinkedTypeShortName());
155 			buffer.append('>');
156 		}
157 		buffer.append(' ');
158 		buffer.append(fieldName);
159 		buffer.append(')');
160 		buffer.append('{');
161 		buffer.append('\n');
162 		buffer.append('\t');
163 		buffer.append('\t');
164 		buffer.append("this");
165 		buffer.append(".");
166 		buffer.append(fieldName);
167 		buffer.append(' ');
168 		buffer.append('=');
169 		buffer.append(' ');
170 		buffer.append(fieldName);
171 		buffer.append(';');
172 		buffer.append('\n');
173 		buffer.append('\t');
174 		buffer.append('}');
175 		buffer.append('\n');
176 		buffer.append('\n');
177 	}

179 	@Override
180 	public void createSetter(ClassVisitor cv, String className, String type, Class<?> c) {
181 
182 		// for jpl 3.x.x compatibility
183 		// not support wrapper class in
184 		// method calling
185 
186 		String supportedType = type;
187 		if (supportedType.equals(Type.getDescriptor(Integer.class))) {
188 			supportedType = Type.getDescriptor(int.class);
189 			String methodName = "set" + getName().substring(0, 1).toUpperCase() + getName().substring(1);
190 			MethodVisitor mv = cv.visitMethod(Opcodes.ACC_PUBLIC, methodName, "(" + supportedType + ")V", null, null);
191 			mv.visitCode();
192 			mv.visitVarInsn(Opcodes.ALOAD, 0);
193 			mv.visitTypeInsn(Opcodes.NEW, Type.getInternalName(Integer.class));
194 			mv.visitInsn(Opcodes.DUP);
195 			mv.visitVarInsn(Opcodes.ILOAD, 1);
196 			mv.visitMethodInsn(Opcodes.INVOKESPECIAL, Type.getInternalName(Integer.class), "<init>",
197 					Type.getMethodDescriptor(Type.VOID_TYPE, Type.getType(int.class)), false);
198 			mv.visitFieldInsn(Opcodes.PUTFIELD, className, getName(), type);
199 			mv.visitInsn(Opcodes.RETURN);
200 			mv.visitMaxs(4, 2);
201 			mv.visitEnd();
202 		} else if (supportedType.equals(Type.getDescriptor(Float.class))) {
203 			supportedType = Type.getDescriptor(float.class);
204 			String methodName = "set" + getName().substring(0, 1).toUpperCase() + getName().substring(1);
205 			MethodVisitor mv = cv.visitMethod(Opcodes.ACC_PUBLIC, methodName, "(" + supportedType + ")V", null, null);
206 			mv.visitCode();
207 			mv.visitVarInsn(Opcodes.ALOAD, 0);
208 			mv.visitTypeInsn(Opcodes.NEW, Type.getInternalName(Float.class));
209 			mv.visitInsn(Opcodes.DUP);
210 			mv.visitVarInsn(Opcodes.FLOAD, 1);
211 			mv.visitMethodInsn(Opcodes.INVOKESPECIAL, Type.getInternalName(Float.class), "<init>",
212 					Type.getMethodDescriptor(Type.VOID_TYPE, Type.getType(float.class)), false);
213 			mv.visitFieldInsn(Opcodes.PUTFIELD, className, getName(), type);
214 			mv.visitInsn(Opcodes.RETURN);
215 			mv.visitMaxs(4, 2);
216 			mv.visitEnd();
217 		} else {
218 			String methodName = "set" + getName().substring(0, 1).toUpperCase() + getName().substring(1);
219 			MethodVisitor mv = cv.visitMethod(Opcodes.ACC_PUBLIC, methodName, "(" + type + ")V", null, null);
220 			mv.visitCode();
221 			mv.visitVarInsn(Opcodes.ALOAD, 0);
222 			mv.visitVarInsn(Type.getType(c).getOpcode(Opcodes.ILOAD), Type.getType(c).getSize());
223 			mv.visitFieldInsn(Opcodes.PUTFIELD, className, getName(), type);
224 			mv.visitInsn(Opcodes.RETURN);
225 			mv.visitMaxs(2, 2);
226 			mv.visitEnd();
227 		}
