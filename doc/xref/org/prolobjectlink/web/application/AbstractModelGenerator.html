90  	private void assertValidClassIndicator(PrologClause clause) {
91  		if (!clause.getIndicator().equals(CLASS)) {
92  			throw new IndicatorError("No valid class descriptor predicate " + clause);
93  		}
94  	}
95  
96  	private void assertValidFieldIndicator(PrologTerm prologTerm) {
97  		String indicator = prologTerm.getIndicator();
98  		if (!indicator.equals(FIELD) && !indicator.equals(KEY) && !indicator.equals(ONETOONE)
99  				&& !indicator.equals(ONETOMANY) && !indicator.equals(MANYTOONE) && !indicator.equals(MANYTOMANY)) {
100 			throw new IndicatorError("No valid field descriptor predicate " + prologTerm);
101 		}
102 	}

104 	public final List<PersistenceUnitInfo> getPersistenceUnits() {
105 		String model = "model.pl";
106 		String database = "database.pl";
107 
108 		Object hbm2ddl = DEFAULT_HBM2DDL;
109 		Object sqlshow = DEFAULT_SQLSHOW;
110 		Object sqlformat = DEFAULT_SQLFORMAT;
111 
112 		List<PersistenceUnitInfo> l = new ArrayList<PersistenceUnitInfo>();
113 		File appRoot = getWebDirectory();
114 		File[] apps = appRoot.listFiles();
115 		for (File file : apps) {
116 
117 			String appName = file.getName();
118 			String appPath = file.getPath();
119 			String separator = File.separator;
120 
121 			String databasePath = appPath + separator + database;
122 			PrologEngine databaseEngine = provider.newEngine(databasePath);
123 			Object jpaProvider = databaseEngine.query("provider(X)").oneResult().get(0);
124 			Object jpaDriver = databaseEngine.query("driver(X)").oneResult().get(0);
125 			Object jpaUser = databaseEngine.query("user(X)").oneResult().get(0);
126 			Object jpaPwd = databaseEngine.query("password(X)").oneResult().get(0);
127 			Object jpaUrl = databaseEngine.query("url(X)").oneResult().get(0);
128 
129 			// for embedded databases
130 			if (jpaUrl.toString().contains(HSQLDBFileDriver.prefix)) {
131 				String rectify = jpaUrl.toString().replace(HSQLDBFileDriver.prefix, "");
132 				if (runOnLinux() || runOnOsX()) {
133 					try {
134 						String str = getDBDirectory().getCanonicalPath();
135 						jpaUrl = new String(HSQLDBFileDriver.prefix + str + "/hsqldb" + rectify)
136 								.replace(File.separatorChar, '/');
137 					} catch (IOException e) {
138 						LoggerUtils.error(getClass(), LoggerConstants.IO, e);
139 					}
140 				} else if (runOnWindows()) {
141 					try {
142 						File[] roots = File.listRoots();
143 						for (File root : roots) {
144 							String str = getDBDirectory().getCanonicalPath();
145 							if (str.startsWith(root.getCanonicalPath())) {
146 								jpaUrl = new String(HSQLDBFileDriver.prefix + str + "/hsqldb" + rectify)
147 										.replace(root.getCanonicalPath(), "/").replace(File.separatorChar, '/');
148 							}
149 						}
150 					} catch (IOException e) {
151 						LoggerUtils.error(getClass(), LoggerConstants.IO, e);
152 					}
153 				}
154 			}
155 
156 			String modelPath = appPath + separator + model;
157 			PrologEngine modelEngine = provider.newEngine(modelPath);
158 			DatabaseSchema schema = new DatabaseSchema(modelPath, provider, factory, owner);
159 
160 			//
161 			String xmlVersion = "1.0";
162 			String xmlEncoding = "UTF-8";
163 			String xmlPersistenceVersion = "2.1";
164 			String xmlPersistenceXmlns = "http://java.sun.com/xml/ns/persistence";
165 			String xmlPersistenceXmlnsXsi = "http://www.w3.org/2001/XMLSchema-instance";
166 			String xmlPersistenceXsiSchemalocation = "http://java.sun.com/xml/ns/persistence http://java.sun.com/xml/ns/persistence/persistence_2_0.xsd";
167 			JPAPersistenceSchemaVersion schemaVersion = new JPAPersistenceSchemaVersion(xmlVersion, xmlEncoding);
168 			JPAPersistenceVersion persistenceVersion = new JPAPersistenceVersion(xmlPersistenceVersion,
169 					xmlPersistenceXmlns, xmlPersistenceXmlnsXsi, xmlPersistenceXsiSchemalocation);
170 			PersistenceUnitTransactionType transactionType = PersistenceUnitTransactionType.RESOURCE_LOCAL;
171 			JPAPersistenceUnitInfo info = new JPAPersistenceUnitInfo(null, schemaVersion, persistenceVersion, appName,
172 					transactionType);
173 
174 			info.setPersistenceProviderClassName("" + jpaProvider + "");
175 			// info.setProperty(DATABASE_ACTION, "" + jpaDatabaseAction + "");
176 			// info.setProperty(SCRIPT_ACTION, "" + jpaScriptAction + "");
177 			info.setProperty(SQLFORMAT, "" + sqlformat + "");
178 			info.setProperty(SQLSHOW, "" + sqlshow + "");
179 			info.setProperty(HBM2DDL, "" + hbm2ddl + "");
180 			info.setProperty(DRIVER, "" + jpaDriver + "");
181 			info.setProperty(USER, "" + jpaUser + "");
182 			info.setProperty(PWD, "" + jpaPwd + "");
183 			info.setProperty(URL, "" + jpaUrl + "");
184 
185 			Map<String, Class<?>> resolved = new HashMap<String, Class<?>>();
186 			Map<String, EntityClass> pending = new HashMap<String, EntityClass>();
187 			Map<String, EntityClass> processed = new HashMap<String, EntityClass>();
188 
189 			Class<?> ppc = JavaReflect.classForName("" + jpaProvider + "");
190 
191 			for (PrologClause clause : modelEngine) {
192 
193 				int position = 0;
194 
195 				assertValidClassIndicator(clause);
196 
197 				PrologTerm head = clause.getHead();
198 				PrologTerm className = head.getArgument(0);
199 				PrologTerm classFields = head.getArgument(2);
200 				String name = (String) converter.toObject(className);
201 				// we use the entity class comment to pass the
202 				// persistence unit name that match with application name
203 				EntityClass entity = new EntityClass(name, appName, schema, ppc);
204 				PrologList fields = (PrologList) classFields;
205 
206 				for (PrologTerm field : fields) {
207 
208 					String ftype = "";
209 					Class<?> c = null;
210 					Class<?> linkedClass = null;
211 					DatabaseField attribute = null;
212 
213 					assertValidFieldIndicator(field);
214 
215 					PrologTerm fieldName = field.getArgument(0);
216 					PrologTerm fieldType = field.getArgument(1);
217 
218 					String fname = (String) converter.toObject(fieldName);
219 
220 					if (fieldType.hasIndicator("set", 1)) {
221 						PrologTerm linkedType = fieldType.getArgument(0);
222 						linkedClass = converter.toClass(linkedType);
223 						c = Set.class;
224 					} else if (fieldType.hasIndicator("list", 1)) {
225 						PrologTerm linkedType = fieldType.getArgument(0);
226 						linkedClass = converter.toClass(linkedType);
227 						c = List.class;
228 					} else if (fieldType.hasIndicator("collection", 1)) {
229 						PrologTerm linkedType = fieldType.getArgument(0);
230 						linkedClass = converter.toClass(linkedType);
231 						c = Collection.class;
232 					} else if (fieldType.hasIndicator("map", 2)) {
233 						PrologTerm linkedKeyType = fieldType.getArgument(0);
234 						PrologTerm linkedValueType = fieldType.getArgument(1);
235 						c = Map.class;
236 					} else {
237 						ftype = (String) converter.toObject(fieldType);
238 						c = converter.toClass(ftype);
239 					}
240 
241 					if (c == null) {
242 
243 						if (linkedClass != null) {
244 							attribute = entity.addField(fname, "", position++, c, linkedClass);
245 						} else {
246 							// we use the entity class comment to pass the persistence unit name
247 							// that match with application name
248 							EntityClass type = new EntityClass(ftype, appName, schema, ppc);
249 							attribute = entity.addField(fname, "", position++, Object.class);
250 							pending.put(fname, type);
251 						}
252 
253 					} else {
254 
255 						if (linkedClass != null) {
256 							attribute = entity.addField(fname, "", position++, c, linkedClass);
257 						} else {
258 							attribute = entity.addField(fname, "", position++, c);
259 						}
260 
261 					}
262 
263 					if (field.hasIndicator("key", 2)) {
264 						attribute.setPrimaryKey(true);
265 					}
266 					if (field.hasIndicator("oneToOne", 2)) {
267 						attribute.setOneToOne(true);
268 						attribute.setOneToMany(false);
269 						attribute.setManyToOne(false);
270 						attribute.setManyToMany(false);
271 					}
272 					if (field.hasIndicator("oneToMany", 2)) {
273 						attribute.setOneToOne(false);
274 						attribute.setOneToMany(true);
275 						attribute.setManyToOne(false);
276 						attribute.setManyToMany(false);
277 					}
278 					if (field.hasIndicator("manyToOne", 2)) {
279 						attribute.setOneToOne(false);
280 						attribute.setOneToMany(false);
281 						attribute.setManyToOne(true);
282 						attribute.setManyToMany(false);
283 					}
284 					if (field.hasIndicator("manyToMany", 2)) {
285 						attribute.setOneToOne(false);
286 						attribute.setOneToMany(false);
287 						attribute.setManyToOne(false);
288 						attribute.setManyToMany(true);
289 					}
290 
291 				}
292 
293 				byte[] bytecode = entity.compile();
294 				Class<?> cls = loader.defineClass(name, bytecode);
295 				info.addManagedClassName(name);
296 				info.addManagedClass(bytecode);
297 				info.addManagedClass(cls);
298 
299 				processed.put(name, entity);
300 				resolved.put(name, cls);
301 
302 			}
303 
304 			for (Entry<String, EntityClass> entry : processed.entrySet()) {
305 				EntityClass entity = entry.getValue();
306 				Collection<DatabaseField> fields = entity.getFields();
307 				for (DatabaseField databaseField : fields) {
308 					if (databaseField.getType() == Object.class) {
309 						String fname = databaseField.getName();
310 						EntityClass ftype = pending.get(fname);
311 						Class<?> cls = resolved.get(ftype.getName());
312 						databaseField.setType(cls);
313 					}
314 				}
315 			}
316 
317 			modelEngine.dispose();
318 			l.add(info);
319 
320 		}
321 
322 		return l;
323 
324 	}
