24  package org.prolobjectlink.web.application;

26  import java.io.File;
27  import java.io.FileInputStream;
28  import java.io.FileNotFoundException;
29  import java.io.FileOutputStream;
30  import java.io.FileWriter;
31  import java.io.IOException;
32  import java.util.List;
33  import java.util.jar.JarEntry;
34  import java.util.jar.JarOutputStream;
35  import java.util.jar.Manifest;
36  
37  import javax.persistence.spi.PersistenceUnitInfo;
38  
39  import org.prolobjectlink.db.jpa.spi.JPAPersistenceUnitInfo;
40  import org.prolobjectlink.logging.LoggerConstants;
41  import org.prolobjectlink.logging.LoggerUtils;
42  
43  public class ModelProcessor extends AbstractWebApplication {
44  
45  	private final ModelGenerator modelGenerator;
46  	private final String temp = System.getProperty("java.io.tmpdir");
47  
48  	public ModelProcessor(ModelGenerator modelGenerator) {
49  		this.modelGenerator = modelGenerator;
50  	}
51  
52  	public void processModel() {
53  
54  		List<PersistenceUnitInfo> units = modelGenerator.getPersistenceUnits();
55  
56  		// write persistence.xml header
57  		StringBuilder builder = new StringBuilder();
58  		builder.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
59  		builder.append('\n');
60  		builder.append("<persistence version=\"1.0\"\r\n"
61  				+ "	xmlns=\"http://java.sun.com/xml/ns/persistence\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"\r\n"
62  				+ "	xsi:schemaLocation=\"http://java.sun.com/xml/ns/persistence http://java.sun.com/xml/ns/persistence/persistence_1_0.xsd\">");
63  		builder.append('\n');
64  
65  		try {
66  
67  			String prolobjectlinkJpxModelJar = "prolobjectlink-jpx-model.jar";
68  			FileOutputStream stream = new FileOutputStream(temp + prolobjectlinkJpxModelJar);
69  			JarOutputStream out = new JarOutputStream(stream, new Manifest());
70  
71  			for (PersistenceUnitInfo unit : units) {
72  				JPAPersistenceUnitInfo jpaUnit = (JPAPersistenceUnitInfo) unit;
73  				jpaUnit.writePersistenceXml(builder);
74  				jpaUnit.writeByteCode(temp);
75  				jpaUnit.jar(out, temp);
76  			}
77  
78  			// write persistence.xml footer
79  			builder.append('\n');
80  			builder.append("</persistence>");
81  
82  			// write to persistence.xml file
83  			String persistenceXml = temp + "persistence.xml";
84  			FileWriter writer = new FileWriter(persistenceXml);
85  			writer.write("" + builder + "");
86  			writer.close();
87  
88  			// copy persistence.xml into .jar
89  			byte[] buffer = new byte[1024];
90  			File file = new File(persistenceXml);
91  			JarEntry jarEntry = new JarEntry("META-INF/" + file.getName());
92  			jarEntry.setTime(file.lastModified());
93  			out.putNextEntry(jarEntry);
94  			FileInputStream in = new FileInputStream(file);
95  			while (true) {
96  				int nRead = in.read(buffer, 0, buffer.length);
97  				if (nRead <= 0)
98  					break;
99  				out.write(buffer, 0, nRead);
100 			}
101 			in.close();
102 			out.close();
103 
104 			// copy from temp directory to lib folder
105 			String jarInLib = getLibDirectory() + File.separator + prolobjectlinkJpxModelJar;
106 			FileInputStream fips = new FileInputStream(temp + prolobjectlinkJpxModelJar);
107 			FileOutputStream fops = new FileOutputStream(jarInLib);
108 			copy(fips, fops);
109 
110 		} catch (FileNotFoundException e) {
111 			LoggerUtils.error(getClass(), LoggerConstants.FILE_NOT_FOUND, e);
112 		} catch (IOException e) {
113 			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
114 		}
115 	}

117 }
