114 	public AbstractResourceLoader() {
115 
116 		for (String resource : resources) {
117 			StringBuilder scriptCode = new StringBuilder();
118 			String path = resource.replace("/io/github/prolobjectlink/web", "");
119 			InputStream stream = getClass().getResourceAsStream(resource);
120 			InputStreamReader reader = new InputStreamReader(stream);
121 			BufferedReader buffer = new BufferedReader(reader);
122 			try {
123 				String line = buffer.readLine();
124 				while (line != null) {
125 					scriptCode.append(line);
126 					line = buffer.readLine();
127 				}
128 			} catch (IOException e) {
129 				LoggerUtils.error(getClass(), LoggerConstants.IO, e);
130 			} finally {
131 				try {
132 					reader.close();
133 				} catch (IOException e) {
134 					LoggerUtils.error(getClass(), LoggerConstants.IO, e);
135 				}
136 				try {
137 					buffer.close();
138 				} catch (IOException e) {
139 					LoggerUtils.error(getClass(), LoggerConstants.IO, e);
140 				}
141 			}
142 			ResourceServlet s = new ResourceServlet("" + scriptCode + "");
143 			ServletUrlMapping m = new ServletUrlMapping(s, path);
144 			mappings.add(m);
145 		}
146 	}
147 
148 	public List<ServletUrlMapping> getMappings() {
149 		return mappings;
150 	}
