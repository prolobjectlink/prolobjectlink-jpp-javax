22  package io.github.prolobjectlink.db.server;
23  
24  import java.io.File;
25  import java.io.FileFilter;
26  import java.io.IOException;
27  
28  import org.hsqldb.Server;
29  import org.hsqldb.persist.HsqlProperties;
30  import org.hsqldb.server.ServerAcl.AclFormatException;
31  
32  import io.github.prolobjectlink.db.DatabaseServer;
33  import io.github.prolobjectlink.db.DatabaseServerType;
34  import io.github.prolobjectlink.logging.LoggerConstants;
35  import io.github.prolobjectlink.logging.LoggerUtils;
36  import io.github.prolobjectlink.web.platform.AbstractWebPlatform;
37  
38  public class HSQLServer extends AbstractWebPlatform implements DatabaseServer {
39  
40  	private final HsqlProperties hsqlProperties = new HsqlProperties();
41  	private final Server server = new Server();
42  
43  	public HSQLServer() {
44  		File dbdir = getDBDirectory();
45  		File[] dbs = dbdir.listFiles();
46  		for (File file : dbs) {
47  			if (file.isDirectory() && file.getName().equals("hsqldb")) {
48  				File[] scripts = file.listFiles(new FileFilter() {
49  
50  					@Override
51  					public boolean accept(File arg0) {
52  						return arg0.getName().endsWith(".script");
53  					}
54  
55  				});
56  				for (int i = 0; i < scripts.length; i++) {
57  					File x = scripts[i];
58  					String name = x.getName().substring(0, x.getName().lastIndexOf(".script"));
59  					hsqlProperties.setProperty("server.database." + i, "file:" + dbdir + "/hsqldb/" + name);
60  					hsqlProperties.setProperty("server.dbname." + i, name);
61  				}
62  			}
63  		}
64  
65  		try {
66  			server.setProperties(hsqlProperties);
67  		} catch (IOException e) {
68  			LoggerUtils.error(getClass(), LoggerConstants.IO, e);
69  		} catch (AclFormatException e) {
70  			LoggerUtils.error(getClass(), LoggerConstants.RUNTIME_ERROR, e);
71  		}
72  		server.setTrace(false);
73  
74  	}
75  
76  	@Override
77  	public void startup() throws IOException, ClassNotFoundException {
78  		server.start();
79  	}
80  
81  	@Override
82  	public DatabaseServerType getType() {
83  		return DatabaseServerType.TCP;
84  	}
85  
86  	@Override
87  	public boolean isRunning() {
88  		return server.getState() == 1 && server.getState() == 4;
89  	}
90  
91  	@Override
92  	public String getName() {
93  		return server.getProductName();
94  	}
95  
96  	@Override
97  	public String getURL() {
98  		return server.getDefaultWebPage();
99  	}
100 
101 	@Override
102 	public void shutdown() {
103 		server.stop();
104 	}
105 
106 }
