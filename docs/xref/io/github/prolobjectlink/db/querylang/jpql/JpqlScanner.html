736 	/**
737 	 * Refills the input buffer.
738 	 *
739 	 * @return <code>false</code>, iff there was new input.
740 	 * 
741 	 * @exception java.io.IOException if any I/O-Error occurs
742 	 */
743 	private boolean zzRefill() throws java.io.IOException {
744 
745 		/* first: make room (if you can) */
746 		if (zzStartRead > 0) {
747 			zzEndRead += zzFinalHighSurrogate;
748 			zzFinalHighSurrogate = 0;
749 			System.arraycopy(zzBuffer, zzStartRead, zzBuffer, 0, zzEndRead - zzStartRead);
750 
751 			/* translate stored positions */
752 			zzEndRead -= zzStartRead;
753 			zzCurrentPos -= zzStartRead;
754 			zzMarkedPos -= zzStartRead;
755 			zzStartRead = 0;
756 		}
757 
758 		/* is the buffer big enough? */
759 		if (zzCurrentPos >= zzBuffer.length - zzFinalHighSurrogate) {
760 			/* if not: blow it up */
761 			char newBuffer[] = new char[zzBuffer.length * 2];
762 			System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
763 			zzBuffer = newBuffer;
764 			zzEndRead += zzFinalHighSurrogate;
765 			zzFinalHighSurrogate = 0;
766 		}
767 
768 		/* fill the buffer with new input */
769 		int requested = zzBuffer.length - zzEndRead;
770 		int totalRead = 0;
771 		while (totalRead < requested) {
772 			int numRead = zzReader.read(zzBuffer, zzEndRead + totalRead, requested - totalRead);
773 			if (numRead == -1) {
774 				break;
775 			}
776 			totalRead += numRead;
777 		}
778 
779 		if (totalRead > 0) {
780 			zzEndRead += totalRead;
781 			if (totalRead == requested) { /* possibly more input available */
782 				if (Character.isHighSurrogate(zzBuffer[zzEndRead - 1])) {
783 					--zzEndRead;
784 					zzFinalHighSurrogate = 1;
785 				}
786 			}
787 			return false;
788 		}
789 
790 		// totalRead = 0: End of stream
791 		return true;
792 	}
